Here is an example of a best practice boilerplate for a Button styled component using the styled-components library:

Copy code
import styled, { css } from 'styled-components';

// Common styles that will be shared by all buttons
const commonStyles = css`
  display: inline-block;
  border: none;
  outline: none;
  border-radius: 4px;
  font-size: 16px;
  line-height: 1;
  text-align: center;
  cursor: pointer;
  transition: all 200ms ease-in-out;

  &:hover {
    // Add hover styles here
  }

  &:active {
    // Add active (pressed) styles here
  }

  &:disabled {
    // Add disabled styles here
  }
`;

// The Button component
const Button = styled.button`
  ${commonStyles}
  // Add specific styles for this button here
`;

export default Button;
